use information_schema;
desc tables;

---
select table_name "Tabla", case when data_length/1024>1 then concat(data_length/1024," Kilob") else concat(data_length, " Bytes")
end as "Tamaño" from tables where table_schem="mysql";


--- Mostrar el tamaño promedio de las tablas, número de tablas y cantidad total de registros de las bases de datos.
 select table_schema "BD", avg(data_length+index_length) "Tamaño promedio",count(*) 
"Total tablas",sum(table_rows) "Cantidad total" from tables group by table_schema;
